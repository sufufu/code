```javascript
/*
 * 目标：
 *   1. 理解字符串索引、length
 *   2. 掌握字符串常用方法
 * */

// 字符串方法: 字符串是被单引号、双引号、反引号(``)包裹的0到多个字符。而且是 js 的一种基本数据类型
// 字符串也有索引，而且它的索引也是浏览器分配的。第一个字符的索引为0，第二个索引为1以此类推 
// 字符串有 length 属性，length 代表字符串中的字符个数，可以通过字符串的索引获取某些字符
// tips：有长度 有索引，就可以用 for 循环遍历

// 声明一个字符串
let str = 'aaa';
let str2 = "bbb";
let str3 = `ccc`;

// 常用方法

// 1. 获取指定索引位置的字符或者指定索引位置的 Unicode 编码

// charAt(索引) 获取指定索引位置的字符
var str5 = '珠峰培训zhufengpeixun';
var char = str5.charAt(16); // str[索引] 也可以获取字符
console.log(char);
console.log(str5.length); // 17

// charCodeAt(索引) 获取字符串中指定索引位置的字符的 Unicode 编码
var uni = strs.charCodeAt(3);
console.log(uni);

// 2. 字符串的截取赋值、拼接

// substr(n, m) 从索引 n 开始截取 m 个，如果不写 m 就是截取到末尾。如果 n 和 m 都不写，从头到尾都截取
var st1 = strs.substr(2, 2);
console.log(st1); // 培训
var st2 = strs.substr(2);
console.log(st2); // 培训 zhufengpeixun
var st3 = strs.substr(); // 珠峰培训 zhufengpeixun
console.log(st3);

// substring(n, m) 从索引 n 开始截取到索引 m（不包含m）；如果不写 m，表示截取到末尾。如果 n 和 m 都不写，就是从头到尾复制一遍
var st4 = strs.substring(4, 7);
console.log(st4);
var st5 = strs.substring(4);
var st6 = strs.substring();
console.log(st5, st6);

// 3. 大小写转换（只能转换英文字母）

// toLowerCae() 大写转小写
var st7 = 'ABC'.toLowerCase();
console.log(st7);

// toUpperCase() 小写转大写
var st8 = 'abc'.toUpperCase();
console.log(st8);

// 4. 获取指定字符在字符串中出现的索引位置
// indexOf() 获取字符在字符串首次出现的位置，如果出现过就返回索引位置，没出现过返回-1；
var st9 = 'abcda'.indexOf('a');

// lastIndexOf() 获取字符在字符串中最后一次出现的位置，如果出现过就返回索引位置，没出现过就返回 -1；
var st10 = 'abcda'.lastIndexOf('a');
console.log(st9, st10);

// includes() 判断字符串中是否出现过某一个字符串，如果出现过返回 true，没出现返回 false 【ES6 新增的，不能在老旧浏览器中使用，IE8 以下】
var st11 = 'abcd'.includes('abc'); // true
console.log(st11);

// 5. split() 把字符串按照指定的分隔符拆分成数组（和数组 join 相反）
var sentence = 'I am a FE, I love programming';
var ary = sentence.split(',');
console.log(ary);
```

let e = str.endsWith("World"); // 返回布尔值 标识参数字符串是否在原字符串尾部

let g = str.repeat(3) // 将字符串重复 3遍
let h = str.padStart(20, "a") // 重复第二个参数来补全 字符串 直到字符串位数为 20 在前面补全
let i = str.padEnd(20, "a") // 重复第二个参数来补全 字符串 直到字符串位数为 20 在后面补全
