```javascript
/*
* 目标：
* 1. 掌握数组常用的方式
* 2. 掌握各个方法参数及、返回值、原数组是否改变
 */

// 数组方法

// 1. 数组的删除和添加

// push
// 作用：向数组末尾增追加项 可以是一项也可以是多项，用逗号分隔
// 参数：要追加的数组项
// 返回值：数组返回追加项后的新长度
// 原数组是否改变：改变

let ary1 = [1, 2, 3, 4, 5, 6];
let r1 = ary1.push(4, 5);
console.log('push：');
console.log(r1);
console.log(ary1);

// pop
// 作用：删除数组末尾一项
// 参数：无
// 返回值：被删除的数组项
// 原数组是否改变:改变

let ary2 = [2, 3, 5];
let r2 = ary2.pop();
console.log('pop：');
console.log(r2);
console.log(ary2);

// unshift
// 作用：向数组开头添加项（可以是一项也可以是多项）
// 参数：要添加到数组开头的项
// 返回值：数组添加项后的新长度
// 原数组是否改变：改变

let ary3 = [2, 3, 5];
let r3 = ary3.unshift(1, 2, 3);
console.log('unshift：');
console.log(r3);
console.log(ary3);

// shift 
// 作用：删除数组开头一项
// 参数：无
// 返回值：被删除的数组项
// 原素组是否改变：改变

let ary4 = [2, 3, 5];
let r4 = ary4.shift();
console.log('shift：');
console.log(r4);
console.log(ary4);

// splice(n, m,);
// 作用：从索引 n 开始，删除 m 个
// 参数：起始索引 n，要删除的个数 m；m 不写，表示删除到最后一项。如果 n 和 m 都不写一个也不删
// 返回值：由被删除的数组成的新数组
// 原数组是否改变：改变

let ary5 = [1, 2, 3, 4, 5, 6];
let r5 = ary5.splice(2, );
console.log('splilce(n, m)：');
console.log(r5);
console.log(ary5);

// splice(n, m, x);
// 作用：从索引 n 开始，删除 m 个，用 x 替换被删除的部分
// 参数：n 为起始索引，m 为删除个数，x 为替换项
// 返回值：由被删除项组成的新数组
// 原数组是否改变：改变

let ary6 = [1, 2, 3, 4, 5, 6];
let r6 = ary6.splice(1, 2, 100);
console.log('splilce(n, m, x)：');
console.log(r6);
console.log(ary6);

// splice(n, 0, x);
// 作用：从索引 n 开始，不删除任何项，把 x 插入到索引 n 的那一项之前
// 参数：n 为起始索引，删除0项，x 为替换项
// 返回值：空数组
// 原素组是否改变：改变

let ary7 = [1, 2, 3, 4, 5, 6];
let r7 = ary7.splice(1, 0, 5)
console.log('splilce(n, x)：');
console.log(r7);
console.log(ary7);

// 2. 数组的复制、拼接

// slice(n, m);
// 作用：从索引 n 开始，复制到索引 m （不包含 m）
// 参数：起始索引 n；结束索引 m； m 不写表示复制到末尾， n 和 m 都不写就是从头复制到末尾
// 返回值：复制出来的项组成的新数组
// 原数组是否改变：不改变

let ary8 = [1, 2, 3, 4, 5, 6];
let r8 = ary8.slice();
console.log('slice(n, x)：');
console.log(r8);
console.log(ary8);

// concat
// 作用：拼接数组
// 参数：要拼接的数组或者数组项
// 返回值：拼接过后的新数组（不是在原数组的基础上直接拼接，而是返回一个新数组）
// 原数组是否改变：不改变

let ary9 = [1, 2, 3, 4, 5, 6];
let r9 = ary9.concat([1, 2, 3, 4, 5, 6]);
console.log('concat：');
console.log(r9);
console.log(ary9);
// 如果 concat 不传参表示将原数组复制一份 
let ary10 = ary9.concat();
// 复制的数组和原数组没关系
console.log(ary9 === ary10); // false

// 3. 数组转字符串

// join 
// 作用：用指定的分隔符，将数组连接成字符串
// 参数：指定分隔符（字符串）
// 返回值：用指定的分隔符连接成的字符串
// 原数组是否改变：不改变

let ary11 = [1, 2, 3, 4, 5, 6];
let r11 = ary11.join('+');
console.log('join：');
console.log(r11);
console.log(ary11);
// eval() 方法把字符串解析成 js 代码执行

// toString
// 作用：把数组转换为字符串
// 参数：无
// 返回值：用逗号连接数组项组成的字符串
// 原数组是否改变：不改变

let ary12 = [1, 2, 3, 4, 5, 6];
let r12 = ary12.toString();
console.log('toString：');
console.log(r12);
console.log(ary12);

// 4. 某一项是否在数组中出现过

// indexOf()
// 作用：获取某一项在数组中第一次出现的索引位置
// 参数：数组项
// 返回值：如果这一项在数组中存在，就返回其第一次出现的索引值，如果没有就返回-1
// 原数组是否改变：不改变

let ary13 = [1, 2, 3, 4, 5, 6];
let r13 = ary13.indexOf(55);
console.log('indexOf：');
console.log(r13);
console.log(ary13);

// lastIndexOf()
// 作用：获取某一项在数组中最后一次出现的索引位置
// 参数：数组项
// 返回值：如果这一项在数组中存在，就返回其最后一次出现的索引值，如果没有就返回-1
// 原数组是否改变：不改变

let ary14 = [1, 2, 3, 4, 5, 6];
let r14 = ary13.lastIndexOf(55);
console.log('lastIndexOf：');
console.log(r14);
console.log(ary14);

// inCludes() 【es6新增的方法，老旧浏览器不能用】
// 作用：判断数组是否包含某一项 
// 参数：数组项
// 返回值：如果包含就返回 true 不包含就返回 false 

let ary15 = [1, 2, 3, 4, 5, 6];
let r15 = ary13.includes(55);
console.log('inCloudes：');
console.log(r15);
console.log(ary15);

// 5. 数组的排序、翻转

// 语法：sort
/**
 * sort(
    function (a, b) {
        return a + b; // 降序
        return a - b; // 升序
        // sort 方法里面不传回调函数时只能排 0 - 9 的数字
        
    })
*/
// 作用：升序或者降序排列数组
// 参数：回调函数(回调函数：当作实参传递给函数执行就叫做回调函数)
// 返回值：排过序的原数组
// 原数组是否发生改变：改变

let ary16 = [1, 2, 8, 4, 9, 5];
let r16 = ary16.sort();
console.log('sort：');
console.log(r16);
console.log(ary16);

// reverse
// 作用：把数组翻转过来排列
// 参数：无
// 返回值：反转过后的原数组
// 原数组是否改变：改变

let ary17 = [1, 2, 8, 4, 9, 5];
let r17 = ary17.reverse();
console.log('reverse：');
console.log(r17);
console.log(ary17);

// 6. 遍历数组

// 语法：forEach(function (item, index) {})

// 作用：遍历数组
// 参数：回调函数；回调函数的形参：item 表示遍历时数组中的每一项，index 遍历时当前项索引；数组有多少项，回调函数就会执行多少回。
// 返回值：无
// 原数组是否改变：取决于回调函数中是否修改原数组，如果修改，原数组就改变，如果不修改就不改变

var ary18 = [1, 3, 5, 7, 9];
ary18.forEach(function(item, index) {

});
console.log('frEach：');
console.log(ary18);

// map
// 语法：map(function (item, index) {})
// 作用：将原数组映射成为一个新数组
// 参数：回调函数，数组由多少项，回调函数就会执行多少回
// 返回值：由回调函数的返回值组成的新数组
// 原数组是否发生改变：不变

let ary19 = [1, 3, 5, 7, 9];
let r19 = ary19.map(function(item, index) {
	return item * 2;
});
console.log('map：');
console.log(r19);
console.log(ary19);
```
