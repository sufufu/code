    <title>数组去重</title>

<body>
    <script>
        /* 1. 基于 es6 的 Set （对应的是 Map）
         *    因为 new Set 自带去重效果，但去重后是个对象
         *    所以把每一项用展开运算符(...)写到数组里 
         */
        let ary1 = [1, 3, 4, 5, 1, 5, 8, 7, 9, 8, 5, 35];
        console.log("Set 类去重：");
        ary1 = [...new Set(ary1)];
        console.log(ary1);

        // Set 也可以用来字符串去重
        console.log('去重字符串');
        console.log([...new Set('ababbc')].join(''));

        /* 2. norepeat 方法实现数组去重
         *    参数：要去重的数组
         *    返回值：去重后的新数组
         */
        let ary2 = [1, 2, 3, 4, 5, 5, 8, 7, 8, 5, 35];
        console.log("删除末尾项去重：");
        function norepeat(ary) {
            // 创建一个空对象
            let obj = {};
            // 循环每一项，然后放到对象里
            for (let i = 0; i < ary.length; i++) {
                // item 存储了每一项
                let item = ary[i];
                // 每一次存之前判断验证 obj 中是否有这一项
                if (obj[item] !== undefined) {
                    // 重复的这一项变成最后一项
                    ary[i] = ary[ary.length - 1];
                    // 删除最后一项
                    ary.length--;
                    i--;
                    // 跳过
                    continue;
                }
                // 对象的每一项就是数组的
                obj[item] = item;
            }
            return ary;
        }
        console.log(norepeat(ary2));

        /* 3. 双 for 去重
         *    遍历数组把数组中的每一项都取出来（不包括最后一项）因为最后一项后面没有项
         *    用取出来的项和齐后面的所有项比较一遍
         *    如果后面的项和取出的项相同就证明重复了，把重复的的一项从原数组中删掉
         */
        let ary3 = [1, 3, 4, 5, 5, 8, 7, 8, 5, 35];
        console.log("双 for 去重：");
        function unique(arr) {
            // for 循环遍历数组，把每一项取出来（不包含最后一项）
            for (let i = 0; i < arr.length - 1; i++) {
                let item = arr[i]; // 每一项
                // 用另一个循环实现当前项和后面所有项比较
                for (let x = i + 1; x < arr.length; x++) {
                    // 判断当前项 item 和 后面项 arr[x] 是否相同，如果相同就删掉
                    if (item === arr[x]) {
                        arr.splice(x, 1);
                        // 这样直接删除会引起数组塌陷。因为删除数组的某一项后，这一项后面的所有项的索引都会减小1，此时如果 x++ 就会跳过被删除项紧邻的后面的那一项。所以先 x-- 然后 x++ 相当于删除后没加没减
                        x--;
                    }
                }
            }
            return arr;
        }
        console.log(unique(ary3));

    </script>
